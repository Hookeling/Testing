#include <pitches.h>  // BUZZER Library
#include <Keypad.h>   // Keypad Library
#include "SafeState.h"  // Library to store Code

// Hardware Pin
#define BUZZER 18

// Konfigurasi Keypad
const byte KEYPAD_ROWS = 4;
const byte KEYPAD_COLS = 4;
byte rowPins[KEYPAD_ROWS] = {13, 12, 14, 27};
byte colPins[KEYPAD_COLS] = {26, 25, 33, 32}; 

char keys[KEYPAD_ROWS][KEYPAD_COLS] = {
  {'1', '2', '3', 'A'},
  {'4', '5', '6', 'B'},
  {'7', '8', '9', 'C'},
  {'*', '0', '#', 'D'}
};

Keypad keypad = Keypad(makeKeymap(keys), rowPins, colPins, KEYPAD_ROWS, KEYPAD_COLS);

// Stores the CODE
SafeState safeState;

// ==========================================
// 1. BASE ACTION FUNCTIONS
// ==========================================

void lock() {
  Serial.println(" -> [HARDWARE ACTION]: Servo moving to LOCKED position.");
  // Letakkan kod untuk servo lock
}

void unlock() {
  Serial.println(" -> [HARDWARE ACTION]: Servo moving to UNLOCKED position.");
  // Letakkan kod untuk servo unlock
}

void showStartupMessage() {
  Serial.println("\n=== HookelSafe v1.0 ===");
  Serial.println("Welcome!");
  delay(500);
  
  Serial.println("Testing Alert Buzzer...");
  tone(BUZZER, 10000);
  delay(500);
  noTone(BUZZER);
}

String inputSecretCode() {
  Serial.print("Enter Code: [____]");
  String result = "";
  
  while (result.length() < 4) {
    char key = keypad.getKey();
    if (key >= '0' && key <= '9') {
      Serial.print('*');
      result += key;
    }
  }
  Serial.println(); 
  return result;
}

void showWaitScreen(int delayMillis) {
  Serial.print("Processing: ");
  for (byte i = 0; i < 10; i++) {
    delay(delayMillis / 10);
    Serial.print("=");
  }
  Serial.println(" Done!");
}

bool setNewCode() {
  Serial.println("Enter your NEW 4-digit code:");
  String newCode = inputSecretCode();

  Serial.println("Confirm your NEW code:");
  String confirmCode = inputSecretCode();

  if (newCode.equals(confirmCode)) {
    safeState.setCode(newCode);
    Serial.println("Password changed successfully!");
    tone(BUZZER, 5000);
    delay(500);
    noTone(BUZZER);
    return true;
  } else {
    Serial.println("Error: Code mismatch!");
    tone(BUZZER, 1000); 
    delay(1000);
    noTone(BUZZER);

    return false;
  }
}

// ==========================================
// 2. LOGIC OPERATIONS
// ==========================================

void safeUnlockedLogic() {
  Serial.println("\n--- SAFE IS UNLOCKED ---");
  Serial.println("Press [#] to Lock | Press [A] to Change Password");

  char key = keypad.getKey();
  while (key != 'A' && key != '#') {
    key = keypad.getKey();
  }

  bool readyToLock = true;
  if (key == 'A') {
    readyToLock = setNewCode();
  }

  if (readyToLock) {
    Serial.println("Locking down container mechanisms...");
    safeState.lock();
    lock();
    showWaitScreen(1000);
  }
}

void safeLockedLogic() {
  Serial.println("\n--- SAFE IS LOCKED ---");
  
  String userCode = inputSecretCode();
  showWaitScreen(1000);

  if (safeState.unlock(userCode)) {
    Serial.println("Access Granted! Welcoming User.");
    tone(BUZZER, 5000);
    delay(500);
    noTone(BUZZER);
    unlock();
  } else {
    Serial.println("ACCESS DENIED! Invalid Credentials.");
    tone(BUZZER, 1000);
    delay(2000);
    noTone(BUZZER);
  }
}

// ==========================================
// EXECUTION BLOCK
// ==========================================

void setup() {  
  Serial.begin(115200);
  delay(1000); 
  
  // ESP32 Core v3.0+ PWM setup
  pinMode(BUZZER, OUTPUT);

  if (safeState.locked()) {
    lock();
  } else {
    unlock();
  }

  showStartupMessage();
}

void loop() {

// Mula

  if (safeState.locked()) {
    safeLockedLogic();
  } else {
    safeUnlockedLogic();
  }
}
